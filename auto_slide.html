[소스코드] https://github.com/codefactory/cfSlider

이 프로그램은 아무런 제약없이 사용하실 수 있습니다. [프로그램 개발문의] master@codefactory.kr
var timer;		// setInterval이 리턴하는 inrevalId

// setInterval을 사용해서 3초에 한번씩  #slider에'cfSliderNext' 이벤트를 발생시켜 주는 함수입니다.
function slideFunc() {
	timer = setInterval(function() {
		$('#slider').trigger('cfSliderNext');	// #slider에서 'cfSliderNext' 이벤트를 발생시키면 다음 슬라이드가 보이게 됩니다. 'cfSliderNext' 라는 이름은 마음대로 바꾸시면 됩니다.
	}, 3000);
}

// cfSlider 만들기
$('#slider').cfSlider({
	display: 5,		// 화면에 5개가 나오도록 CSS 설정을 해놨을 경우 display 옵션에 5를 주셔야 합니다. #slider의 width는 li하나의 width 곱하기 5로 CSS에서 설정해주세요
	nextEventType: 'cfSliderNext',	// 다음 슬라이드가 보여지게 하는데 사용할 커스텀 이벤트 입니다. 위에 있는 slideFunc 함수 안에서 이 이벤트를 발생시키고 있습니다.
	prevEventType: 'cfSliderPrev'	// 이전 슬라이드가 보여지게 하는데 사용할 커스텀 이벤트 입니다. 이 데모에서는 이전 보기는 사용하고 있지 않지만 필요할 경우 이렇게 설정해서 쓰면 됩니다.
});

slideFunc();	// 페이지가 로딩되면 slideFunc를 실행시켜서 3초에 한번씩 슬라이드 되게 합니다.

$('#slider').hover(function() {
	clearInterval(timer);	// 사용자가 마우스를 #slider 위에 놓으면 슬라이드를 멈춥니다.
}, function() {
	slideFunc();			// 마우스를 빼면 슬라이드를 시작합니다.
});
